 // Word Counter
 const wordCount = (sentence) => {
     let wordsArr = sentence.replace(/[.|,]/g,"").split(" ");
     let answerList = {};

     wordsArr.forEach(i=> (!answerList[i]) ? (answerList[i] = 1) : (answerList[i]++))
  
     return Object.entries(answerList)
     .sort(([keyA,valueA], [keyB,valueB]) => (valueB - valueA))
     .reduce((acc, [key, value]) => ({ ...acc, [key]: value }), {});
}
